#version 430
uniform mat4 matModel;
uniform mat4 matView;
uniform mat4 matProj;

layout (location=0) in  vec3 position;
layout (location=1) in  vec3 normal;
layout (location=2) in  vec2 uv;

out vec3 WorldPos_GS_in;                                                                        
out vec3 Normal_GS_in; 
out vec2 TexCoord_GS_in;

void main(void) {
    gl_Position = matProj*matView*matModel*vec4(position.x*0.2, position.y*0.2, position.z*0.2, 1.0); 
	
	vec4 temp= matView*vec4(position, 1.0f); 
	WorldPos_GS_in= vec3(temp.x, temp.y, temp.z);
	
	temp= transpose(inverse(matView))*vec4(normal, 1.0f);
	Normal_GS_in = vec3(temp.x, temp.y, temp.z);	
	
	TexCoord_GS_in = uv;
}
